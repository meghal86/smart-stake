Troubleshooting

If you see "function set_primary_wallet does not exist":
Run the migration file in Supabase SQL Editor: 
-- File: supabase/migrations/20260120000000_set_primary_wallet_function.sql
CREATE OR REPLACE FUNCTION set_primary_wallet(
  p_user_id UUID,
  p_wallet_address TEXT
)
RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  UPDATE user_wallets
  SET is_primary = FALSE,
      updated_at = NOW()
  WHERE user_id = p_user_id;
  
  UPDATE user_wallets
  SET is_primary = TRUE,
      updated_at = NOW()
  WHERE user_id = p_user_id
    AND LOWER(address) = LOWER(p_wallet_address);
END;
$$;

GRANT EXECUTE ON FUNCTION set_primary_wallet(UUID, TEXT) TO authenticated;
